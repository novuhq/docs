## Basic Example

```typescript
import { Workflow } from '@novu/node';

const welcomeWorkflow = new Workflow({
  name: 'Welcome Flow',
  description: 'Send welcome messages to new users',
  steps: [
    {
      type: 'email',
      template: {
        subject: 'Welcome to {{appName}}!',
        content: 'Hi {{name}}, welcome aboard!'
      }
    },
    {
      type: 'in-app',
      template: {
        content: 'Welcome! Check out our getting started guide.'
      }
    }
  ]
});
```

## Framework Components

### Workflows

Workflows define the notification flow:

```typescript
const workflow = new Workflow({
  name: 'Order Status',
  trigger: 'order.status.changed',
  steps: [/* ... */]
});
```

### Steps

Steps are individual actions in a workflow:

```typescript
const steps = [
  {
    type: 'email',
    template: {/* ... */}
  },
  {
    type: 'delay',
    duration: '1h'
  },
  {
    type: 'push',
    template: {/* ... */}
  }
];
```

### Templates

Templates define the content structure:

```typescript
const template = {
  subject: 'Order #{{orderId}} Update',
  content: `
    Your order status: {{status}}
    {{#if estimated_delivery}}
    Estimated delivery: {{estimated_delivery}}
    {{/if}}
  `
};
```

## Integration Examples

### Next.js

```typescript
// app/api/notifications/route.ts
import { Novu } from '@novu/node';

const novu = new Novu(process.env.NOVU_API_KEY);

export async function POST(req: Request) {
  const { userId, orderId } = await req.json();
  
  await novu.trigger('order-update', {
    to: userId,
    payload: { orderId }
  });
}
```

### Express

```typescript
import express from 'express';
import { Novu } from '@novu/node';

const app = express();
const novu = new Novu(process.env.NOVU_API_KEY);

app.post('/notifications', async (req, res) => {
  const { userId, orderId } = req.body;
  
  await novu.trigger('order-update', {
    to: userId,
    payload: { orderId }
  });
  
  res.sendStatus(200);
});
```
