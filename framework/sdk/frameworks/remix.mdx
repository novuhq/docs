sidebarTitle: "Remix"

```tsx
import { serve } from "@novu/framework/remix";
import { myWorkflow } from "../novu/workflows";

const handler = serve({
    workflows: [myWorkflow],
  });
  
export { handler as action, handler as loader };
```
