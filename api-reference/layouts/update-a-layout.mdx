import { Novu } from '@novu/node';

  description: "Organisation's first layout",
  name: "First Layout",
  identifier: "firstlayout",
  variables: [
    {

      required: true
      defValue: ""
    }
  ]

}
await novu.layouts.update("layoutId", payloadToUpdate);
